import java.util.*;

class Solution {
    public int[] solution(String today, String[] terms, String[] privacies) {
        int tDate = toDays(today);

        // ì•½ê´€ ê¸°ê°„ ì €ì¥
        Map<String, Integer> term = new HashMap<>();
        for (String str : terms) {
            String[] s = str.split(" ");
            term.put(s[0], Integer.parseInt(s[1]));
        }

        ArrayList<Integer> ans = new ArrayList<>();
        for (int i = 0; i < privacies.length; i++) {
            String[] s = privacies[i].split(" ");
            int start = toDays(s[0]);
            int expire = start + term.get(s[1]) * 28; // í•œ ë‹¬ = 28ì¼ ê¸°ì¤€

            if (expire <= tDate) ans.add(i + 1);
        }

        return ans.stream().mapToInt(Integer::intValue).toArray();
    }

    // ë‚ ì§œë¥¼ "YYYY.MM.DD" â†’ ì´ ì¼ìˆ˜ë¡œ ë³€í™˜
    static int toDays(String date) {
        String[] str = date.split("\\.");
        int y = Integer.parseInt(str[0]);
        int m = Integer.parseInt(str[1]);
        int d = Integer.parseInt(str[2]);
        return (y * 12 * 28) + (m * 28) + d;
    }
}
